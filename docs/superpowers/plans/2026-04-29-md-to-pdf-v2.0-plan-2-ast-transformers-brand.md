```python
# src/mdpdf/markdown/transformers/base.py
from abc import ABC, abstractmethod
from typing import List
from mdpdf.markdown.ast import Document

class Transformer(ABC):
    """Base class for AST transformation passes."""

    @abstractmethod
    def run(self, doc: Document) -> Document:
        """Transform the document AST in place. Return the (possibly mutated) document."""
        pass

class TransformerRegistry:
    """Ordered registry of transformers to run in sequence."""

    def __init__(self):
        self.transformers: List[Transformer] = []

    def add(self, transformer: Transformer) -> "TransformerRegistry":
        """Register a transformer. Returns self for chaining."""
        self.transformers.append(transformer)
        return self

    def run_all(self, doc: Document) -> Document:
        """Run all transformers in order. Each transformer receives output of the previous."""
        for transformer in self.transformers:
            doc = transformer.run(doc)
        return doc
```

```python
# src/mdpdf/markdown/transformers/frontmatter.py
import re
from typing import Any, Dict
import yaml
from mdpdf.markdown.ast import Document, RawHtml
from mdpdf.markdown.transformers.base import Transformer

class FrontmatterStripper(Transformer):
    """Extract YAML/TOML frontmatter from document start and populate metadata."""

    def run(self, doc: Document) -> Document:
        """Strip frontmatter if present as first node."""
        if not doc.children:
            return doc

        first = doc.children[0]
        
        # Check if first node is RawHtml containing frontmatter markers (---)
        if isinstance(first, RawHtml):
            content = first.content.strip()
            if content.startswith("---") and "---" in content[3:]:
                # Extract the YAML between the markers
                lines = content.split("\n")
                end_idx = None
                for i in range(1, len(lines)):
                    if lines[i].strip() == "---":
                        end_idx = i
                        break
                
                if end_idx is not None:
                    yaml_content = "\n".join(lines[1:end_idx])
                    try:
                        metadata = yaml.safe_load(yaml_content) or {}
                        doc.metadata.update(metadata)
                        # Remove frontmatter from children
                        doc.children = doc.children[1:]
                    except yaml.YAMLError:
                        # If YAML parsing fails, leave document unchanged
                        pass

        return doc
```

```python
from typing import List, Optional

@dataclass
class OutlineEntry:
    """Single table-of-contents entry."""
    level: int                          # 1-6 for h1-h6
    text: str                           # heading text
    anchor: str                         # auto-generated anchor (h1-heading-text)
    children: List["OutlineEntry"] = field(default_factory=list)

@dataclass
class Outline:
    """Complete table of contents."""
    entries: List[OutlineEntry] = field(default_factory=list)
    
    def to_dict(self) -> dict:
        """Convert to nested dict for serialization."""
        def entry_to_dict(e: OutlineEntry) -> dict:
            return {
                "level": e.level,
                "text": e.text,
                "anchor": e.anchor,
                "children": [entry_to_dict(child) for child in e.children],
            }
        return {"entries": [entry_to_dict(e) for e in self.entries]}
```

```python
# src/mdpdf/markdown/transformers/heading.py
from mdpdf.markdown.ast import Document, Heading, OutlineEntry, Outline
from mdpdf.markdown.transformers.base import Transformer

class OutlineCollector(Transformer):
    """Build table of contents from heading hierarchy."""

    def run(self, doc: Document) -> Document:
        """Walk document and build outline from Heading nodes."""
        outline = Outline()
        stack = []  # Stack of (level, entry) tuples
        
        def add_entry(level: int, text: str, anchor: str):
            entry = OutlineEntry(level=level, text=text, anchor=anchor)
            
            # Pop stack until we find parent level
            while stack and stack[-1][0] >= level:
                stack.pop()
            
            if stack:
                # Add as child of current top
                stack[-1][1].children.append(entry)
            else:
                # Add as top-level entry
                outline.entries.append(entry)
            
            stack.append((level, entry))
        
        for node in doc.children:
            if isinstance(node, Heading):
                # Extract text from heading children
                text = self._extract_text(node)
                anchor = self._make_anchor(text)
                add_entry(node.level, text, anchor)
        
        doc.metadata["outline"] = outline
        return doc

    @staticmethod
    def _extract_text(heading: Heading) -> str:
        """Extract plain text from heading children."""
        # For now, assume heading has a text attribute set by parser
        return getattr(heading, 'text', 'Untitled')

    @staticmethod
    def _make_anchor(text: str) -> str:
        """Convert heading text to URL-safe anchor."""
        import re
        anchor = text.lower()
        anchor = re.sub(r'[^a-z0-9\s-]', '', anchor)
        anchor = re.sub(r'\s+', '-', anchor)
        return anchor
```

```python
# src/mdpdf/brand/schema.py
from typing import Optional, Dict, List, Any
from pydantic import BaseModel, Field, validator

class ThemeConfig(BaseModel):
    """Brand theme (colors, fonts, sizing)."""
    colors: Dict[str, str] = Field(default_factory=dict, description="Color overrides (e.g., primary, accent)")
    fonts: Dict[str, str] = Field(default_factory=dict, description="Font assignments (e.g., heading, body)")
    line_height: float = Field(default=1.5, ge=1.0, le=3.0)
    font_size_base: int = Field(default=12, ge=8, le=20)

    class Config:
        extra = "allow"  # Allow additional fields

class ComplianceConfig(BaseModel):
    """Brand compliance (watermarks, footers, disclaimers)."""
    footer_text: Optional[str] = None
    show_page_numbers: bool = True
    show_render_date: bool = True
    watermark_min_level: str = Field(default="L1", pattern="^L[0-5]$")
    watermark_template: str = "{brand_name} // {user} // {render_date}"
    disclaimer: Optional[str] = None

    class Config:
        extra = "allow"

class Brand(BaseModel):
    """Complete brand pack definition."""
    id: str = Field(..., description="Brand identifier (alphanumeric + hyphens)")
    name: str = Field(..., description="Brand display name")
    schema_version: str = Field(default="1.0.0", description="SemVer version")
    description: Optional[str] = None
    theme: ThemeConfig
    compliance: ComplianceConfig
    assets_dir: Optional[str] = None  # Relative path to assets (logo, fonts, etc.)
    
    # Override policy
    allows_inline_override: bool = True
    allowed_override_fields: Optional[List[str]] = None  # Whitelist; if set, only these allowed
    forbidden_override_fields: List[str] = Field(default_factory=list)  # Always rejected

    @validator("id")
    def validate_id(cls, v):
        import re
        if not re.match(r"^[a-z0-9-]+$", v):
            raise ValueError("id must be lowercase alphanumeric + hyphens")
        return v

    class Config:
        use_enum_values = True
```

```python
# src/mdpdf/brand/safe_paths.py
from pathlib import Path
from typing import Union

class PathTraversalError(ValueError):
    """Raised when a path escapes the intended base directory."""
    pass

def safe_join(base: Path, relative: Union[str, Path]) -> Path:
    """
    Safely join a base path with a relative path, preventing directory traversal.
    
    Raises PathTraversalError if:
    - relative is absolute
    - relative contains .. or similar traversal
    - relative is a file:// URL
    """
    base = Path(base).resolve()
    relative_str = str(relative)
    
    # Reject obvious attack patterns
    if relative_str.startswith("/"):
        raise PathTraversalError(f"Absolute path not allowed: {relative_str}")
    if relative_str.startswith("file://"):
        raise PathTraversalError(f"file:// URLs not allowed: {relative_str}")
    if ".." in relative_str:
        raise PathTraversalError(f"Parent directory traversal not allowed: {relative_str}")
    
    # Join and verify result is still within base
    result = (base / relative_str).resolve()
    
    # Check that result is within base (no symlink escape)
    try:
        result.relative_to(base)
    except ValueError:
        raise PathTraversalError(f"Path escapes base directory: {relative_str} -> {result}")
    
    return result
```

```python
# src/mdpdf/brand/registry.py
from pathlib import Path
from typing import Optional
import yaml
from mdpdf.brand.schema import Brand

class BrandRegistry:
    """3-layer brand pack registry: project > user > built-in."""

    def __init__(
        self,
        project_layer: Optional[Path] = None,
        user_layer: Optional[Path] = None,
        builtin_layer: Optional[Path] = None,
    ):
        self.project_layer = project_layer
        self.user_layer = user_layer
        self.builtin_layer = builtin_layer

    def resolve(self, brand_id: str) -> Optional[Brand]:
        """
        Resolve a brand by ID, checking layers in order: project > user > builtin.
        Returns None if not found.
        """
        for layer in [self.project_layer, self.user_layer, self.builtin_layer]:
            if layer is None:
                continue
            brand = self._load_from_layer(layer, brand_id)
            if brand:
                return brand
        return None

    @staticmethod
    def _load_from_layer(layer_dir: Path, brand_id: str) -> Optional[Brand]:
        """Load a brand from a specific layer directory."""
        brand_path = layer_dir / brand_id / "brand.yaml"
        
        if not brand_path.exists():
            return None
        
        try:
            with open(brand_path) as f:
                data = yaml.safe_load(f)
            return Brand(**data)
        except Exception:
            return None

    def list_available(self) -> dict[str, Path]:
        """Return dict of {brand_id: path} for all available brands."""
        available = {}
        for layer in [self.project_layer, self.user_layer, self.builtin_layer]:
            if layer is None or not layer.exists():
                continue
            for brand_dir in layer.iterdir():
                if brand_dir.is_dir():
                    available[brand_dir.name] = brand_dir
        return available
```

```python
# src/mdpdf/fonts/manager.py
from pathlib import Path
from typing import Dict, Optional, List

class FontManager:
    """Manage font registration and lookup."""

    def __init__(self):
        self.fonts: Dict[str, Dict[str, str]] = {}  # {family: {style: path}}

    def register_bundled(self, bundled_dir: Path) -> None:
        """Register fonts from bundled directory."""
        bundled_dir = Path(bundled_dir)
        
        # Look for font.yaml manifest in each font subdirectory
        for font_dir in bundled_dir.iterdir():
            if not font_dir.is_dir():
                continue
            
            # For now, assume font directory names are font families
            family = font_dir.name
            style_files = {}
            
            # Find .ttf files and assign to styles
            for ttf_file in font_dir.glob("*.ttf"):
                if "regular" in ttf_file.name or ttf_file.stem == family:
                    style_files["regular"] = str(ttf_file)
                elif "bold" in ttf_file.name:
                    style_files["bold"] = str(ttf_file)
                elif "italic" in ttf_file.name:
                    style_files["italic"] = str(ttf_file)
                elif "bolditalic" in ttf_file.name or "bolditalic" in ttf_file.name:
                    style_files["bolditalic"] = str(ttf_file)
            
            if style_files:
                self.fonts[family] = style_files

    def register_user(self, user_dir: Path) -> None:
        """Register fonts from user directory."""
        # Same logic as bundled for now
        self.register_bundled(user_dir)

    def get_font(self, family: str) -> Optional[Dict[str, str]]:
        """Get font files for a family (returns {style: path} dict)."""
        return self.fonts.get(family)

    def list_available(self) -> List[str]:
        """List all registered font families."""
        return list(self.fonts.keys())
```
